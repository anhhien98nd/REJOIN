-- 🌸 Auto TP1 Lock 3 Giai Đoạn
local Players = game:GetService("Players")
local p = Players.LocalPlayer
local STATE = { tp1 = false }

-- ⚙️ Hàm Lock vị trí
local function startLock(pos)
    STATE.tp1 = true
    print("[AUTO] 🔒 Bắt đầu lock vị trí tại:", pos)

    local look
    pcall(function()
        look = p.Character and p.Character:FindFirstChild("HumanoidRootPart")
            and p.Character.HumanoidRootPart.CFrame.LookVector
    end)

    task.spawn(function()
        while STATE.tp1 do
            task.wait()
            local c = p.Character
            if c then
                local hrp = c:FindFirstChild("HumanoidRootPart")
                if hrp then
                    hrp.Velocity = Vector3.zero
                    if look then
                        hrp.CFrame = CFrame.new(pos, pos + look)
                    else
                        hrp.CFrame = CFrame.new(pos)
                    end
                end
            end
        end
    end)
end

-- ⚙️ Hàm tắt lock
local function stopLock()
    if STATE.tp1 then
        STATE.tp1 = false
        print("[AUTO] 🔓 Dừng lock vị trí.")
    end
end

task.spawn(function()
    local pos = Vector3.new(40, 3.63, -130.14)

    -- 🕒 Giai đoạn 1: bật sau 10s và tắt sau 77s
    task.wait(10)
    startLock(pos)
    task.wait(67) -- chạy thêm cho đủ tới 77s (77 - 10)
    stopLock()
    print("[AUTO] ✅ Kết thúc giai đoạn 1 tại 77s")

    -- 🕒 Giai đoạn 2: bật sau 89s và tắt sau 151s
    task.wait(12) -- chờ đến 89s (89 - 77)
    startLock(pos)
    print("[AUTO] 🔁 Bật lại TP1 Lock tại giây 89")
    task.wait(62) -- chạy thêm cho đủ tới 151s (151 - 89)
    stopLock()
    print("[AUTO] ✅ Tắt TP1 Lock lần 2 tại giây 151")

    -- 🕒 Giai đoạn 3: bật sau 176s và tắt sau 228s
    task.wait(25) -- chờ đến 176s (176 - 151)
    startLock(pos)
    print("[AUTO] 🔁 Bật lại TP1 Lock tại giây 176")
    task.wait(52) -- chạy thêm cho đủ tới 228s (228 - 176)
    stopLock()
    print("[AUTO] ✅ Tắt TP1 Lock lần 3 tại giây 228")
end)
