-- üå∏ Auto TP2 Lock 3 Giai ƒêo·∫°n (B·∫£n ho√†n ch·ªânh fix l·ªói)
local Players = game:GetService("Players")
local p = Players.LocalPlayer
local STATE = { tp2 = false }

-- ‚öôÔ∏è Ch·ªù HumanoidRootPart
local function waitForHRP()
    local chr = p.Character or p.CharacterAdded:Wait()
    local hrp = chr:FindFirstChild("HumanoidRootPart")
    local waited = 0
    while not hrp and waited < 5 do
        task.wait(0.2)
        waited += 0.2
        chr = p.Character or p.CharacterAdded:Wait()
        hrp = chr:FindFirstChild("HumanoidRootPart")
    end
    return hrp
end

-- ‚öôÔ∏è H√†m Lock v·ªã tr√≠
local function startLock(pos)
    if STATE.tp2 then return end
    STATE.tp2 = true
    print("[AUTO] üîí B·∫Øt ƒë·∫ßu lock v·ªã tr√≠ t·∫°i:", tostring(pos))

    local hrp = waitForHRP()
    local look
    if hrp then
        pcall(function()
            look = hrp.CFrame.LookVector
        end)
    end

    task.spawn(function()
        while STATE.tp2 do
            task.wait()
            local c = p.Character
            if c then
                local h = c:FindFirstChild("HumanoidRootPart")
                if h then
                    pcall(function()
                        -- ‚ö†Ô∏è D√πng Vector3.new(0,0,0) thay cho Vector3.zero
                        h.Velocity = Vector3.new(0,0,0)
                        if look then
                            h.CFrame = CFrame.new(pos, pos + look)
                        else
                            h.CFrame = CFrame.new(pos)
                        end
                    end)
                end
            end
        end
    end)
end

-- ‚öôÔ∏è H√†m t·∫Øt lock
local function stopLock()
    if STATE.tp2 then
        STATE.tp2 = false
        print("[AUTO] üîì D·ª´ng lock v·ªã tr√≠.")
    end
end

-- üß† Tr√¨nh t·ª± auto 3 giai ƒëo·∫°n
task.spawn(function()
    local pos = Vector3.new(42.29, 3.63, -132.49)

    -- üïí Giai ƒëo·∫°n 1: b·∫≠t sau 10s v√† t·∫Øt sau 77s
    task.wait(10)
    startLock(pos)
    task.wait(67)
    stopLock()
    print("[AUTO] ‚úÖ K·∫øt th√∫c giai ƒëo·∫°n 1 t·∫°i 77s")

    -- üïí Giai ƒëo·∫°n 2: b·∫≠t sau 89s v√† t·∫Øt sau 151s
    task.wait(12)
    startLock(pos)
    print("[AUTO] üîÅ B·∫≠t l·∫°i TP2 Lock t·∫°i gi√¢y 89")
    task.wait(62)
    stopLock()
    print("[AUTO] ‚úÖ T·∫Øt TP2 Lock l·∫ßn 2 t·∫°i gi√¢y 151")

    -- üïí Giai ƒëo·∫°n 3: b·∫≠t sau 176s v√† t·∫Øt sau 228s
    task.wait(25)
    startLock(pos)
    print("[AUTO] üîÅ B·∫≠t l·∫°i TP2 Lock t·∫°i gi√¢y 176")
    task.wait(52)
    stopLock()
    print("[AUTO] ‚úÖ T·∫Øt TP2 Lock l·∫ßn 3 t·∫°i gi√¢y 228")
end)
