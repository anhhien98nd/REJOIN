-- 🌀 Auto Teleport Cổng 1 (teleport 2 lần, sau 3s và sau 83s kể từ bắt đầu)
local Players = game:GetService("Players")
local p = Players.LocalPlayer

local teleportPos = Vector3.new(18.86713409423828, 0.37274569272994995, -114.86202239990234)

local function teleportToGate()
	local chr = p.Character or p.CharacterAdded:Wait()
	local hrp = chr:FindFirstChild("HumanoidRootPart")
	if hrp then
		hrp.CFrame = CFrame.new(teleportPos + Vector3.new(0, 5, 0))
		print("[AUTO] Đã dịch chuyển tới Cổng 1 tại:", os.date("%X"))
	else
		warn("[AUTO] Không tìm thấy HumanoidRootPart — không thể dịch chuyển.")
	end
end

task.spawn(function()
	-- 🔹 Lần 1: sau 3 giây
	task.wait(3)
	print("[AUTO] Dịch chuyển tới Cổng 1 (lần 1 sau 3 giây)")
	teleportToGate()

	-- 🔹 Lần 2: sau thêm 80 giây (tổng 83 giây từ đầu)
	task.wait(80)
	print("[AUTO] Dịch chuyển lại tới Cổng 1 (lần 2 sau 83 giây)")
	teleportToGate()
end)
