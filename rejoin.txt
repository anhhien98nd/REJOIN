-- 🌸 Rejoin Script (Chỉ dành cho acc "ikkopolconloli")

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local p = Players.LocalPlayer

if not p then
    warn("[Rejoin] ❌ Không tìm thấy LocalPlayer.")
    return
end

if p.Name ~= "ikkopolconloli" then
    warn("[Rejoin] 🚫 Bỏ qua — chỉ acc 'ikkopolconloli' mới được rejoin.")
    return
end

-- 🕒 Hẹn Rejoin sau 228 giây
print("[Rejoin] ⏳ Sẽ rejoin sau 228 giây cho tài khoản:", p.Name)

task.delay(228, function()
    print("[Rejoin] 🔁 Đang rejoin tài khoản:", p.Name)

    -- Giãn 1 chút để tránh crash
    task.wait(1.5)

    pcall(function()
        local placeId = game.PlaceId
        local jobId = game.JobId
        TeleportService:TeleportToPlaceInstance(placeId, jobId, p)
    end)

    print("[Rejoin] ✅ Đã gửi yêu cầu rejoin server hiện tại cho:", p.Name, "lúc:", os.date("%X"))
end)
