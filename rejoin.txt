-- 🌸 Auto Rejoin Script (chỉ dành cho acc "ikkopolconloli")

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local player = Players.LocalPlayer or Players.PlayerAdded:Wait()

-- ===== CONFIG =====
local TARGET_NAME = "ikkopolconloli" -- chỉ acc này mới được rejoin
local AUTO_REJOIN_TIME = 154 -- thời gian tự động rejoin (giây)
-- ===================

-- ♻️ Hàm rejoin
local function rejoinGame()
	if player.Name ~= TARGET_NAME then
		warn("[AutoRejoin] 🚫 Bỏ qua — chỉ tài khoản '" .. TARGET_NAME .. "' được phép rejoin.")
		return
	end

	print(("[AutoRejoin] 🔁 Đang rejoin %s tại %s"):format(player.Name, os.date("%X")))
	pcall(function()
		TeleportService:Teleport(game.PlaceId, player)
	end)
end

-- 🕒 Tự động rejoin sau 154 giây
task.delay(AUTO_REJOIN_TIME, function()
	print(("[AutoRejoin] ⏳ Sẽ tự động rejoin sau %d giây..."):format(AUTO_REJOIN_TIME))
	rejoinGame()
end)
