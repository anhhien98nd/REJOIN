-- 🌸 Rejoin Script (Chỉ dành cho acc "tênngườichơi")

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local p = Players.LocalPlayer

if not p then
    warn("[Rejoin] ❌ Không tìm thấy LocalPlayer.")
    return
end

if p.Name ~= "tênngườichơi" then
    warn("[Rejoin] 🚫 Bỏ qua — chỉ acc 'tênngườichơi' mới được rejoin.")
    return
end

-- Thông báo trước khi rejoin
print("[Rejoin] 🔁 Đang rejoin tài khoản:", p.Name)

-- Giãn 1 chút để tránh crash
task.wait(1.5)

pcall(function()
    local placeId = game.PlaceId
    local jobId = game.JobId
    TeleportService:TeleportToPlaceInstance(placeId, jobId, p)
end)

print("[Rejoin] ✅ Đã gửi yêu cầu rejoin server hiện tại cho:", p.Name)
