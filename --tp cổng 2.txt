-- 🌀 Auto Teleport Cổng 2 (3 lần: sau 3s, 63s, 113s)
local Players = game:GetService("Players")
local p = Players.LocalPlayer

-- ⚙️ Toạ độ Cổng 2
local teleportPos = Vector3.new(18.37334884643555, 0.37274569272994995, -140.72775268554688)

-- 🧩 Hàm chờ HumanoidRootPart sẵn sàng
local function waitForHRP()
	local chr = p.Character or p.CharacterAdded:Wait()
	local hrp = chr:FindFirstChild("HumanoidRootPart")
	local waited = 0
	while not hrp and waited < 5 do
		task.wait(0.2)
		waited += 0.2
		chr = p.Character or p.CharacterAdded:Wait()
		hrp = chr:FindFirstChild("HumanoidRootPart")
	end
	return hrp
end

-- 🚀 Hàm dịch chuyển đến Cổng 2
local function teleportToGate()
	local hrp = waitForHRP()
	if hrp then
		pcall(function()
			hrp.CFrame = CFrame.new(teleportPos + Vector3.new(0, 5, 0))
		end)
		print("[AUTO] ✨ Đã dịch chuyển tới Cổng 2 tại:", os.date("%X"))
	else
		warn("[AUTO] ⚠️ Không tìm thấy HumanoidRootPart — không thể dịch chuyển.")
	end
end

-- 🧠 Chuỗi tự động 3 lần
task.spawn(function()
	print("[AUTO] ⏳ Bắt đầu chuỗi Auto Teleport Cổng 2...")

	-- 🔹 Lần 1: sau 3 giây
	task.wait(3)
	print("[AUTO] 🚀 Dịch chuyển tới Cổng 2 (Lần 1 - sau 3s)")
	teleportToGate()

	-- 🔹 Lần 2: sau thêm 60 giây (tổng 63s)
	task.wait(60)
	print("[AUTO] 🚀 Dịch chuyển lại tới Cổng 2 (Lần 2 - sau 63s)")
	teleportToGate()

	-- 🔹 Lần 3: sau thêm 50 giây (tổng 113s)
	task.wait(50)
	print("[AUTO] 🚀 Dịch chuyển lần 3 tới Cổng 2 (sau 113s)")
	teleportToGate()

	print("[AUTO] ✅ Hoàn tất chuỗi dịch chuyển 3 lần Cổng 2.")
end)
